// Firestore Security Rules - Geçici Development İçin
// Bu kurallar sadece development aşamasında kullanılmalı

rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    // Sites koleksiyonu - tüm kullanıcılar okuyabilir
    match /sites/{document} {
      allow read, write: if request.auth != null;
    }
    
    // Team members koleksiyonu - sadece admin yazabilir
    match /team_members/{document} {
      allow read: if request.auth != null;
      allow write: if request.auth != null && 
        (resource == null || resource.data.role == 'admin' || 
         request.auth.token.role == 'admin');
    }
    
    // Users koleksiyonu
    match /users/{document} {
      allow read: if request.auth != null;
      allow write: if request.auth != null && 
        (request.auth.uid == document || request.auth.token.role == 'admin');
    }
    
    // Site assignments
    match /site_assignments/{document} {
      allow read, write: if request.auth != null;
    }
    
    // Daily production data
    match /daily_production/{document} {
      allow read: if request.auth != null;
      allow write: if request.auth != null;
    }
    
    // Diğer koleksiyonlar için genel kural
    match /{document=**} {
      allow read, write: if request.auth != null;
    }
  }
}
